Public API Endpoints
All endpoints below are mounted under /api.

- GET /api/zad-website-document

    listZadWebsiteDocuments
  - Used by:
    D:\JannatBooking\zad_ssr\lib\api.js

- Recommended Nginx site content for the sudo step.
  server {
      listen 80;
      server_name zadhotels.jannatbooking.com;

      client_max_body_size 100M;

      location / {
          proxy_pass http://127.0.0.1:3105;
          proxy_http_version 1.1;
          proxy_set_header Upgrade $http_upgrade;
          proxy_set_header Connection "upgrade";
          proxy_set_header Host $host;
          proxy_set_header X-Real-IP $remote_addr;
          proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
          proxy_set_header X-Forwarded-Proto https;
          proxy_cache_bypass $http_upgrade;
      }
  }
